DHXPlanner.templates.week_date

The date in the header of the view.

Available views: week, weekAgenda 

planner.templates.week_date = function(date1, date2){
	return planner.templates.day_date(date1)+" &ndash; "+planner.templates.
day_date(planner.date.add(date2,-1,"day"));
};

DHXPlanner.templates.timeline_date

The date in the header of the view.

Available views: timeline 

planner.templates.timeline_date = function(date1, date2){
	if (date1.getDay()==date2.getDay() && date2-date1<(24*60*60*1000))
			return planner.templates.day_date(date1);
		return planner.templates.week_date(date1, date2); 
};
